316 class Ref {
317 public:
318  Ref(bool suppress = false) : _suppress(suppress) {
319  static_assert(std::is_base_of<Object, T>::value, "Ref class must be derived from Object!");
320  }
321 
322  std::shared_ptr<T> operator()() { //get
323  if (_empty) {
324  if (!_suppress) Debug::Error("Object Reference", "Reference is null!");
325  return nullptr;
326  }
327  else if (_object.expired()) {
328  if (!_suppress) Debug::Error("Object Reference", "Reference is deleted!");
329  return nullptr;
330  }
331  else return _object.lock();
332  }
333  std::shared_ptr<T> operator->() { return this->operator()(); }
334 
335  void operator()(const std::shared_ptr<T>& ref) { //set
336  _object = ref;
337  _empty = false;
338  }
339  void operator()(const Ref<T>& ref) {
340  _object = ref._object;
341  _empty = false;
342  }
343  void operator()(const T* ref) {
344  _object = ref->shared_from_this();
345  _empty = false;
346  }
347 
348  bool operator ==(const Ref<T>& rhs) const {
349  return this->_object.lock() == rhs._object.lock();
350  }
351 
352  bool operator !=(const Ref<T>& rhs) const {
353  return this->_object.lock() != rhs._object.lock();
354  }
355 
356  void clear() {
357  _empty = true;
358  }
359 
360  bool ok() {
361  return !(_empty || _object.expired());
362  }
363 
364  T* raw() {
365  if (ok()) return _object.lock().get();
366  else return nullptr;
367  }

688  Debug::Warning("ComputeBuffer", "Set: Unable to map buffer!");
689  }
690  memcpy(tar, src, size);
691  glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
692  glBindBuffer(GL_SHADER_STORAGE_BUFFER, 0);
693  return (T*)tar;
