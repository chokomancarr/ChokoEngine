264 template <class T> class Ref {
265 public:
266  Ref(bool suppress = false) : _suppress(suppress) {
267  static_assert(std::is_base_of<Object, T>::value, "Ref class must be derived from Object!");
268  }
269  Ref(std::shared_ptr<T> ref, bool suppress = false) : _suppress(suppress), _object(ref), _empty(false) {
270  static_assert(std::is_base_of<Object, T>::value, "Ref class must be derived from Object!");
271  }
272 
273  std::shared_ptr<T> operator()() { //get
274  if (_empty) {
275  if (!_suppress) Debug::Error("Object Reference", "Reference is null!");
276  return nullptr;
277  }
278  else if (_object.expired()) {
279  if (!_suppress) Debug::Error("Object Reference", "Reference is deleted!");
280  return nullptr;
281  }
282  else return _object.lock();
283  }
284  std::shared_ptr<T> operator->() { return this->operator()(); }
285 
286  void operator()(const std::shared_ptr<T>& ref) { //set
287  _object = ref;
288  _empty = false;
289  }
290  void operator()(const Ref<T>& ref) {
291  _object = ref._object;
292  _empty = false;
293  }
294  void operator()(const T* ref) {
295  if (!ref) clear();
296  else {
297  _object = get_shared<T>((Object*)ref);
298  _empty = false;
299  }
300  }
301  operator bool() const {
302  return !(_empty || _object.expired());
303  }
304 
305  bool operator ==(const Ref<T>& rhs) const {
306  return this->_object.lock() == rhs._object.lock();
307  }
308  bool operator !=(const Ref<T>& rhs) const {
309  return this->_object.lock() != rhs._object.lock();
310  }
311 
312  void clear() {
313  _empty = true;
314  }
315  bool ok() {
316  return !(_empty || _object.expired());
317  }
318  T* raw() {
319  if (ok()) return _object.lock().get();
320  else return nullptr;
321  }
322 
323 private:
324  std::weak_ptr<T> _object;
325  bool _empty = true, _suppress = false;
326 };
